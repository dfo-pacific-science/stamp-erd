Table event {
  event_id varchar [pk]
  parent_event_id varchar [ref: > event.event_id]
  protocol_id varchar [ref: > protocol.protocol_id]  
  event_type varchar // e.g. "sampling", "shipping", "dissection"
  event_date varchar
}

Table event_assertion {
  assertion_id varchar       [pk]
  event_id varchar           [ref: > event.event_id]
  assertion_type varchar     // e.g. "biomass", "shippingMethod"
  assertion_value_numeric numeric
  assertion_unit varchar
}


Table occurrence {
  occurrence_id varchar [pk]
  event_id varchar      [ref: > event.event_id]
  organismID varchar
  organism_quantity varchar
  organism_quantity_type varchar
}


Table material {
  material_entity_id varchar [pk]
  parent_material_id varchar [ref: > material.material_entity_id]
  occurrence_id varchar      [ref: > occurrence.occurrence_id]
  protocol_id varchar        [ref: > protocol.protocol_id]  
  material_type varchar      // e.g. "voucher" or "tissue"
  material_subtype varchar   // e.g. "fin clip"
}

Table material_identifier {
  identifier varchar
  material_entity_id varchar [ref: > material.material_entity_id]
  identifier_type varchar     // e.g. "containerID"
}

Table material_assertion {
  assertion_id varchar        [pk]
  material_entity_id varchar  [ref: > material.material_entity_id]
  assertion_type varchar      // e.g. "priority", "resultsAvailable"
  assertion_value varchar     // e.g. "high", "true"
  assertion_value_numeric numeric
  assertion_unit varchar
}

Table protocol {
  protocol_id varchar [pk]
  protocol_name varchar
  protocol_description varchar
}
